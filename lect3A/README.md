```
(define 1-to-4 (list 1 2 3 4))
(define nil '())
```

```
(define (scale s l)
  (if (null? l)
    nil
    (cons (* s (car l))
          (scale s (cdr l)))))
```

```
(scale 10 1-to-4)
```

```
(define (map f l)
  (if (null? l)
    nil
    (cons (f (car l))
          (map f (cdr l)))))
```

```
(define (scale s l)
  (map (lambda (x) (* s x))
       l))
```

```
(define (square x) (* x x))
(map square 1-to-4)
```

```
(map (lambda (x) (+ x 10)) 1-to-4)
```

```
(define (foreach f l)
  (if (null? l)
      nil
      (cons (f (car l))
            (foreach f (cdr l)))))
```
